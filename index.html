Combat System Framework


local Combat = {}
Combat.__index = Combat

function Combat.new(player)
    local self = setmetatable({}, Combat)
    self.Player = player
    self.Combo = 0
    self.LastAttack = 0
    return self
end

function Combat:Attack()

    local now = tick()

    if now - self.LastAttack > 1 then
        self.Combo = 0
    end

    self.Combo += 1
    self.LastAttack = now

    return self.Combo
end

return Combat

Optimized DataStore Handler


local DataStoreService = game:GetService("DataStoreService")
local PlayerData = DataStoreService:GetDataStore("PlayerData")

local DataHandler = {}

function DataHandler:Load(player)

    local success,data = pcall(function()
        return PlayerData:GetAsync(player.UserId)
    end)

    if success and data then
        return data
    else
        return {
            Coins = 0,
            Level = 1
        }
    end
end

function DataHandler:Save(player,data)

    pcall(function()
        PlayerData:SetAsync(player.UserId,data)
    end)

end

return DataHandler

local Remote = game.ReplicatedStorage:WaitForChild("Attack")

Remote.OnServerEvent:Connect(function(player,action)

    if typeof(action) ~= "string" then return end

    if action == "Punch" then

        local character = player.Character
        if not character then return end

        local humanoid = character:FindFirstChild("Humanoid")
        if not humanoid then return end

        humanoid:TakeDamage(10)

    end

end)
