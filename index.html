CombatSystem.lua
LUA
Combat System Framework
OOP-based combo tracking system with timing validation
local Combat = {}
Combat.__index = Combat

function Combat.new(player)
  local self = setmetatable({}, Combat)
  self.Player = player
  self.Combo = 0
  self.LastAttack = 0
  return self
end

function Combat:Attack()
  local now = tick()
  if now - self.LastAttack > 1 then
    self.Combo = 0
  end
  self.Combo += 1
  self.LastAttack = now
  return self.Combo
end

return Combat
DataHandler.lua
LUA
Optimized DataStore Handler
Safe async data loading with pcall error handling
local DS = game:GetService("DataStoreService")
local Store = DS:GetDataStore("PlayerData")
local Handler = {}

function Handler:Load(player)
  local ok, data = pcall(function()
    return Store:GetAsync(player.UserId)
  end)
  if ok and data then
    return data
  end
  return { Coins = 0, Level = 1 }
end

function Handler:Save(player, data)
  pcall(function()
    Store:SetAsync(player.UserId, data)
  end)
end

return Handler

  if typeof(action) ~= "string" then
    return -- reject invalid types
  end

  if action == "Punch" then
    local char = player.Character
    if not char then return end
    local hum = char:

      hum:TakeDamage(10)

  end
